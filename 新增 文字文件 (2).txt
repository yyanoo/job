
        public static int? Total(int? req)
        {
            using (var context = new AMSEntities())
            {
                var query = context.Contract.AsQueryable();
                query = query.Where(c => context.ContractParty
                .Any(cp => cp.ContractID == c.ContractID && cp.PartyID == req));
                return query.Count();
            }
        }
